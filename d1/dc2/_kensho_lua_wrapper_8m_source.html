257 NSObject* popValue(lua_State* L, int stackLevel)
258 {
259  // Now figure out the datatype of the assignment and get the data out
260  if(lua_isboolean(L, stackLevel))
261  {
262  return @(lua_toboolean(L, stackLevel));
263  }
264  else if(lua_isnumber(L, stackLevel))
265  {
266  return @(lua_tonumber(L, stackLevel));
267  }
268  else if(lua_isstring(L, stackLevel))
269  {
270  const char* tempString = lua_tostring(L, stackLevel);
271  return [NSString stringWithUTF8String:tempString];
272  }
273  else if(lua_isnoneornil(L, stackLevel))
274  {
275  return nil;
276  }
277  else if(lua_istable(L, stackLevel))
278  {
279  // This is a bit of a special case here. We assume this is a wrapper for an NSObject.
280  // This could actually be a new table, in which case this will fail.
281  lua_getfield(L, stackLevel, "__self");
282  NSObject* value = (__bridge id)(lua_touserdata(L, -1));
283 
284  // If this wasn't one of our wrapped objects, pop the table into a NSDictionary map.
285  if(value == nil)
286  {
287 
288  // We'll create an NSDictionary wrapper here
289  NSMutableDictionary* dictionary = [[NSMutableDictionary alloc] init];
290 
291  // stackLevel-1 is now the table we attempted to get
292  while (lua_next(L, stackLevel-1) != 0) {
293  /* uses 'key' (at index -2) and 'value' (at index -1) */
294  printf("%s\n", lua_typename(L, lua_type(L, -2)));
295  if(lua_isstring(L, -2))
296  {
297  const char* tempString = lua_tostring(L, -2);
298  printf("%s\n", tempString);
299 
300  }
301  // Make sure the key is a valid string too!
302  const char* nameRequest = lua_tostring(L, -2);
303  if(nameRequest == NULL)
304  {
305  return 0;
306  }
307 
308  NSObject* subvalue = popValue(L, -1);
309  dictionary[[NSString stringWithUTF8String:nameRequest]] = subvalue;
310 
311  /* removes 'value'; keeps 'key' for next iteration */
312  lua_pop(L, 1);
313  printf("%s\n", lua_typename(L, lua_type(L, -1)));
314  if(lua_isstring(L, -1))
315  {
316  const char* tempString = lua_tostring(L, -1);
317  printf("%s\n", tempString);
318  }
319  }
320 
321  value = [NSDictionary dictionaryWithDictionary:dictionary];
322  }
323  else
324  {
325  lua_pop(L, 1);
326  }
327  /*printf("%s\n", lua_typename(L, lua_type(L, -1)));
328  if(lua_isstring(L, -1))
329  {
330  const char* tempString = lua_tostring(L, -1);
331  printf("%s\n", tempString);
332 
333  }*/
334  return value;
335  }

351 {
352  // Stack is [table, key, value]
353 
354  // Verify that the key is a string
355  if(!lua_isstring(L, -2))
356  {
357  return 0;
358  }
359 
360  // Make sure that the context is a table. This should actually be the global table.
361  if(!lua_istable(L, -3))
362  {
363  return 0;
364  }
365 
366  // Make sure the key is a valid string too!
367  const char* nameRequest = lua_tostring(L, -2);
368  if(nameRequest == NULL)
369  {
370  return 0;
371  }

374  // And then get its __wrapper object, which points to the owning KenshoLuaWrapper.
375  lua_getfield(L, -3, "__wrapper");
376  id target = (__bridge id)(lua_touserdata(L, -1));

382

385  KenshoLuaWrapper* wrapper = target;
386  NSObject* value = nil;
387 
388  value = popValue(L, -1);
389  // Ok, we save this value to the 'parameters' object
390  [wrapper.parameters setValue:value forKey:propertyName];
391 
392  return 1;

471 int callValue(lua_State* L)
472 {
473  NSObject* parameter = popValue(L, -lua_gettop(L));
474  if(![parameter isKindOfClass:ObservablePropertyReference.class])
475  {
476  return 0;
477  }
478  ObservablePropertyReference* reference = (ObservablePropertyReference*)parameter;
479 
480  id object = reference.owner;

482  {
483  object = [object context];
484  }
485

488  if([object respondsToSelector:selector])

495  pushLuaValue(L, result);

500  selector = NSSelectorFromString(selectorName);

505  pushLuaValue(L, result);
