123 int lookupKey(lua_State* L)
124 {
125  // Stack is [tabble, key]
126  if(!lua_isstring(L, -1))
127  {
128  return 0;
129  }
130 
131  if(!lua_istable(L, -2))
132  {
133  return 0;
134  }
135 
136  // Get the pointer to the context object
137  lua_getfield(L, -2, "__self");
138  id target = (__bridge id)(lua_touserdata(L, -1));
139  if(target == nil)

160 NSObject* popValue(lua_State* L, int stackLevel)
161 {
162  // Now figure out the datatype of the assignement and get the data out
163  if(lua_isboolean(L, stackLevel))
164  {
165  return @(lua_toboolean(L, stackLevel));
166  }
167  else if(lua_isnumber(L, stackLevel))
168  {
169  return @(lua_tonumber(L, stackLevel));
170  }
171  else if(lua_isstring(L, stackLevel))
172  {
173  const char* tempString = lua_tostring(L, stackLevel);
174  return [NSString stringWithUTF8String:tempString];
175  }
176  else if(lua_istable(L, stackLevel))
177  {
178  // This is a bit of a special case here. We assume this is a wrapper for an NSObject.
179  // This could actually be a new table, in which case this will fail.
180  lua_getfield(L, stackLevel, "__self");
181  NSObject* value = (__bridge id)(lua_touserdata(L, -1));
182  if(value == nil)
183  {
184 
185  // We'll create an NSDictionary wrapper here
186  NSMutableDictionary* dictionary = [[NSMutableDictionary alloc] init];
187 
188  while (lua_next(L, stackLevel-1) != 0) {
189  /* uses 'key' (at index -2) and 'value' (at index -1) */
190  printf("%s\n", lua_typename(L, lua_type(L, -2)));
191  if(lua_isstring(L, -2))
192  {
193  const char* tempString = lua_tostring(L, -2);
194  printf("%s\n", tempString);
195 
196  }
197  // Make sure the key is a valid string too!
198  const char* nameRequest = lua_tostring(L, -2);
199  if(nameRequest == NULL)
200  {
201  return 0;
202  }
203 
204  NSObject* subvalue = popValue(L, -1);
205  dictionary[[NSString stringWithUTF8String:nameRequest]] = subvalue;
206 
207  /* removes 'value'; keeps 'key' for next iteration */
208  lua_pop(L, 1);
209  printf("%s\n", lua_typename(L, lua_type(L, -1)));
210  if(lua_isstring(L, -1))
211  {
212  const char* tempString = lua_tostring(L, -1);
213  printf("%s\n", tempString);
214 
215  }
216  }
217 
218  value = [NSDictionary dictionaryWithDictionary:dictionary];
219  }
220  printf("%s\n", lua_typename(L, lua_type(L, -1)));
221  if(lua_isstring(L, -1))
222  {
223  const char* tempString = lua_tostring(L, -1);
224  printf("%s\n", tempString);
225 
226  }
227  return value;
228  }
229 
230  return nil;
231 }

243 int lookupNewKey(lua_State* L)
244 {
245  // Stack is [table, key, value]
246 
247  // Verify that the key is a string
248  if(!lua_isstring(L, -2))
249  {
250  return 0;
251  }
252 
253  // Make sure that the context is a table. This should actually be the global table.
254  if(!lua_istable(L, -3))
255  {
256  return 0;
257  }
258 
259  // Make sure the key is a valid string too!
260  const char* nameRequest = lua_tostring(L, -2);
261  if(nameRequest == NULL)
262  {
263  return 0;
264  }
265 
266  // Get the pointer to the context table (the global table here)
267  // And then get it's __wrapper object, which points to the owning KenshoLuaWrapper.
268  lua_getfield(L, -3, "__wrapper");
269  id target = (__bridge id)(lua_touserdata(L, -1));
270  lua_pop(L, 1);
271  if(target == nil)
272  {
273  return 0;
274  }
275 
276  // Convert to an NSString and start navigating the Objective-C side
277  NSString* propertyName = [NSString stringWithUTF8String:nameRequest];
278  KenshoLuaWrapper* wrapper = target;
279  NSObject* value = nil;
280 
281  value = popValue(L, -1);
282  // Ok, we save this value to the 'parameters' object
283  [wrapper.parameters setValue:value forKey:propertyName];
284 
285  return 1;
286 }
