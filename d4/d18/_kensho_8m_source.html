123 - (void) bindToView:(UIView *)view context:(KenshoContext*)initialContext
124 {
125  KenshoContext* context = initialContext;
126 
127  // verify the format ..... label: .....
128  if(view.ken.allKeys.count > 0)
129  {
130  for(NSString* bindType in view.ken.allKeys)
131  {
132  NSString* bindValue = view.ken[bindType];
133 
134  KenshoLuaWrapper* wrapper = [[KenshoLuaWrapper alloc] initWithKensho:self context:context code:bindValue];
135 
136  // Now, definitively, the observable passed into any given binding is either a value type,
137  // or an NSDictionary of observables/values for multiple parameters
138  NSObject<IObservable>* targetValue = [[Computed alloc] initWithKensho:self
139  calculator:^NSObject *(NSObject<IObservable> *o) {
140  return [wrapper evaluate:bindValue];
141  }];
142 
143 
144 
145  // Register this binding against this view
146  NSString* viewKey = [NSString stringWithFormat:@"%p", view];
147  if(assignedBindings[viewKey] == nil)
148  {
149  assignedBindings[viewKey] = [NSMutableSet set];
150  }
151  [assignedBindings[viewKey] addObject:targetValue];
152 
153 
154  // Handle the special-case structural bindings
155  if([bindValue isEqualToString:@"with"])
156  {
157  // We will need to change the context from now on
158  // This is an observable... so \todo rebind if this changes
159  context = [[KenshoContext alloc] initWithContext:targetValue.value parent:context];
160  continue;
161  }
162 
163  NSObject<Binding>* binding = nil;
164 
165  // iterate through classes and their parents
166  Class viewClass = nil;
167  while((viewClass = (viewClass == nil ? view.class : [viewClass superclass])))
168  {
169  // We dispatch based on the view type and the binding type here
170  if(bindingFactories[viewClass] == nil)
171  {
172  continue;
173  }
174 
175  bindingFactoryMethod method = bindingFactories[viewClass][bindType];
176  if(method == nil)
177  {
178  continue;
179  }
180 
181  binding = method(view,bindType,targetValue,context);
182  if(binding == nil)
183  {
184  continue;
185  }
186 
187  // Register this binding against this view
188  NSString* viewKey = [NSString stringWithFormat:@"%p", view];
189  if(assignedBindings[viewKey] == nil)
190  {
191  assignedBindings[viewKey] = [NSMutableSet set];
192  }
193  [assignedBindings[viewKey] addObject:binding];
194 
195  [binding updateValue];
196 
197  break;
198  }
199 
200  if(binding == nil)
201  {
202  NSLog(@"Error - could not create binding from factory for type %@ of class %@",
203  bindType,
204  NSStringFromClass(view.class));
205  }
206 
207  }
208  }
209 
210  for(UIView* subview in view.subviews)
211  {
212  [self bindToView:subview context:context];
213  }
214 }
