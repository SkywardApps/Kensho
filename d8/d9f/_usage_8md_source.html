220 - (void) testAutomaticViewBinding
221 {
222  // Get the application kensho manager
223  Kensho* ken = [[Kensho alloc] init];
224 
225  Person* person = [[Person alloc] initWithKensho:ken];
226  person.firstName.value = @"Roger";
227  person.lastName.value = @"Moore";
228 
229  // Get the main view controller
230  UIViewController* mainController = [[[[UIApplication sharedApplication] delegate] window] rootViewController];
231  UIView* rootView = mainController.view;
232 
233  [ken applyBindings:rootView viewModel:person];
234 
235  // Find all the views we care about and make sure they got the right values
236  UITextField* firstNameField = [rootView viewWithTag:1];
237  XCTAssertEqualObjects(@"Roger", firstNameField.text);
238 
239  UITextField* lastNameField = [rootView viewWithTag:2];
240  XCTAssertEqualObjects(@"Moore", lastNameField.text);
241 
242  UILabel* fullNameLabel = [rootView viewWithTag:3];
243  XCTAssertEqualObjects(@"Hello, Roger Moore", fullNameLabel.text);
244 
245  // Update the first name and emit the event
246  firstNameField.text = @"Bobbie";
247  [firstNameField sendActionsForControlEvents:UIControlEventEditingChanged];
248 
249  // And make sure this 'user input' propogated as needed!
250  XCTAssertEqualObjects(@"Bobbie", person.firstName.value);
251  XCTAssertEqualObjects(@"Hello, Bobbie Moore", fullNameLabel.text);
252 
253 }
